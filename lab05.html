# Create the function.
getmode <- function(v) {
   uniqv <- unique(v)
   uniqv[which.max(tabulate(match(v, uniqv)))]
}

my_median <- function(x){
  sort(x)
  n = length(x)
  med_point <- (n+1)/2
      if(n %% 2 ==0){
        absn = abs(med_point)
        absn1 = n+1
        med <- (x[absn] +x[absn1])/2
    }else{
      med <- x[med_point]
    }
  return (med)
 
}

my_sd <- function(x){
  x_bar = mean(x)
  n = length(x)
  x2 = 0
  for( i in x ){
    x2 = x2+ (i-x_bar)^2
  } 
  sd = sqrt(x2/(n-1))
  return(sd)
}
